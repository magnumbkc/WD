            function check(){
            let age = parseInt(document.getElementById("age").value);
            let msg;
            if (age >= 18){
                msg="YOU are eligible";
            }
            else {
                msg="you are not eligible";
            }
            document.getElementById("display").innerHTML=msg;
            return false;
        }
