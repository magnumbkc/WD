        $total = $m1 + $m2 + $m3;

        $per = ($total / 300) * 100;

        if($per >= 75){

            $grade = "A";
        }

        else if($per >= 60){

            $grade = "B";
        }

        else if($per >= 45){

            $grade = "C";
        }

        else if($per >= 35){

            $grade = "D";
        }

        else{

            $grade = "F";
        }

        echo "

        <table border='1'>

            <tr>
                <th>Total</th>
                <th>Percentage</th>
                <th>Grade</th>
            </tr>

            <tr>
                <td>$total</td>
                <td>$per%</td>
                <td>$grade</td>
